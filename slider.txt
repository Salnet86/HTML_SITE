<body>
    <!-- Contenitore dello slider -->
    <div class="slider-container">
        <!-- Wrapper che contiene tutte le immagini -->
        <div class="slider-wrapper">
            <img src="https://picsum.photos/id/1011/400/300" alt="Immagine 1">
            <img src="https://picsum.photos/id/1012/400/300" alt="Immagine 2">
            <img src="https://picsum.photos/id/1015/400/300" alt="Immagine 3">
        </div>

        <!-- Bottoni navigazione -->
        <button class="slider-btn prev">&#10094;</button>
        <button class="slider-btn next">&#10095;</button>
    </div>

    <script>
        // Prendo il wrapper e le immagini
        const wrapper = document.querySelector('.slider-wrapper');
        const images = document.querySelectorAll('.slider-wrapper img');

        // Bottoni
        const prevBtn = document.querySelector('.prev');
        const nextBtn = document.querySelector('.next');

        // Indice dell'immagine corrente
        let index = 0;

        // Funzione per aggiornare la posizione dello slider
        function showSlide(i) {
            // Controllo per ciclicit√† (loop)
            if (i < 0) {
                index = images.length - 1;
            } else if (i >= images.length) {
                index = 0;
            } else {
                index = i;
            }
            // Traslazione del wrapper in base all'indice
            wrapper.style.transform = `translateX(${-index * 100}%)`;
        }

        // Eventi bottoni
        prevBtn.addEventListener('click', () => {
            showSlide(index - 1);
        });

        nextBtn.addEventListener('click', () => {
            showSlide(index + 1);
        });

        // Optional: auto-scroll ogni 5 secondi
        // setInterval(() => {
        //     showSlide(index + 1);
        // }, 5000);
    </script>
