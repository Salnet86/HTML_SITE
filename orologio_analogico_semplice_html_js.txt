<body>
<canvas id="clock" width="300" height="300"></canvas>
<script>
const canvas = document.getElementById("clock");
const ctx = canvas.getContext("2d");
const cx = canvas.width / 2;
const cy = canvas.height / 2;
const raggio = 100;
const lunghezzaTacca = 20;

// Disegno le 60 tacche
for (let i = 0; i < 60; i++) {
  const ang = (i * 6) * Math.PI / 180;
  const x1 = cx + raggio * Math.sin(ang);
  const y1 = cy - raggio * Math.cos(ang);
  const x2 = cx + (raggio - lunghezzaTacca) * Math.sin(ang);
  const y2 = cy - (raggio - lunghezzaTacca) * Math.cos(ang);

  ctx.beginPath();
  ctx.lineWidth = (i % 5 === 0) ? 3 : 1;
  ctx.moveTo(x1, y1);
  ctx.lineTo(x2, y2);
  ctx.stroke();
}

// Disegno i numeri 1-12
const r_num = raggio - 35; // distanza dal centro per i numeri
ctx.font = "20px Arial";
ctx.textAlign = "center";
ctx.textBaseline = "middle";

for (let i = 1; i <= 12; i++) {
  const ang = i * 30 * Math.PI / 180;
  const x = cx + r_num * Math.sin(ang);
  const y = cy - r_num * Math.cos(ang);
  ctx.fillText(i.toString(), x, y);
}

// Cerchio centrale di riferimento
ctx.beginPath();
ctx.arc(cx, cy, 3, 0, 2 * Math.PI);
ctx.fillStyle = "red";
ctx.fill();
</script>
