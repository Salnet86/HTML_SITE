  <script>
    /* --------------------------------------------------
       JAVASCRIPT SLIDER AUTOMATICO
       Questo script fa scorrere automaticamente le tre immagini dello slider.
       Ogni 3 secondi, l’indice dell’immagine aumenta e la funzione sposta il gruppo di immagini a sinistra.
       Quando arriva all’ultima immagine, torna alla prima (ciclo infinito).
    -------------------------------------------------- */

    let index = 0; // Indice dell'immagine corrente (parte da 0)
    const slides = document.getElementById('slides'); // Riferimento al contenitore delle immagini

    function nextSlide() {
      index = (index + 1) % 3; // Incrementa l’indice e torna a 0 dopo la terza immagine
      slides.style.transform = `translateX(-${index * 100}%)`; // Sposta il gruppo immagini a sinistra
    }

    // Esegue la funzione ogni 3 secondi per far scorrere le immagini automaticamente
    setInterval(nextSlide, 3000);
  </script>
