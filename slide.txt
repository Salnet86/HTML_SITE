    top: 0;
    left: 0;
    opacity: 0;
    transition: opacity 1s;
  }

  .slides img.active {
    opacity: 1;
  }

    position: absolute;
    top: 50%;

    background-color: rgba(0,0,0,0.5);
    color: white;

    <img src="https://picsum.photos/id/1037/600/400">
  </div>

  <button class="prev">❮</button>
  <button class="next">❯</button>
</div>

<script>
var slides = document.getElementsByClassName('slides')[0].getElementsByTagName('img');
var prev = document.getElementsByClassName('prev')[0];
var next = document.getElementsByClassName('next')[0];
var size = 0; // indice corrente

// funzione che gestisce il fade e il controllo con if
function cambiaSlide() {
  for (var i = 0; i < slides.length; i++) {
    slides[i].className = '';
  }
  slides[size].className = 'active';
}

// Pulsante avanti
next.onclick = function() {
  if (size < slides.length - 1) {
    size++; // size++
  } else {
    size = 0; // torna all'inizio
  }
  cambiaSlide();
}

// Pulsante indietro
prev.onclick = function() {
  if (size > 0) {
    size--; // size--
  } else {
    size = slides.length - 1; // torna all'ultima
  }
  cambiaSlide();
}

// Mostra immagine iniziale
cambiaSlide();
</script>
